% Commandes communes

% Pour ajouter le plus simplement du monde une image
% @syntaxe
%   \image{path/to/image}
% Le path est relatif par rapport au dossier "img" situé à la
% racine du livre (au même niveau que les dossiers sources_md,
% assets_latex, etc.)
% Pour le NOM DE L'IMAGE, ne pas mettre d'underscore ou de
% caractères spéciaux.
% Le fait de NE PAS METTRE D'EXTENSION permet de choisir le
% meilleur format (jpg, ps, btm, etc.)
\newcommand{\image}[1]{\includegraphics{../../img/#1}}
