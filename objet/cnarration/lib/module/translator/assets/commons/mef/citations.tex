% Mise en forme pour les citations et références
%


% Apparence des "\cite{<clé du film>}"
\DeclareLabelname[movie]{
  \field{title} \field{year}
}

% Apparence d'une référence interne (i.e. dans le livre courant)
% Le premier paramètre est le contenu de \ref, le second paramètre est le
% contenu de \pageref
\newcommand{\intref}[1]{%
  \ref{#1}, p.\pageref{#1}%
}

% Apparence pour une référence externe (i.e. dans un autre livre de la collection)
% Argument 1 : le titre du livre
% Argument 2 : le \ref (la référence)
% Note : la page n'est pas fournie, puisque les livres seront écrits de façon 
% asynchrone, seule la table des matières sera figée au moment de l'impression du
% premier livre.
\newcommand{\extrefpart}[2]{%
\bookserie{#1}, Partie \ref{#2} %
}
\newcommand{\extrefchapter}[2]{%
\bookserie{#1}, Chapitre \ref{#2} %
}
\newcommand{\extrefsection}[2]{%
\bookserie{#1}, Chapitre \ref{#2} %
}
